INSERT INTO PRODUCT(product_name, product_type, unit_price, modified_date)
  SELECT   src.product_name,
         src.product_type,
         src.unit_price,
         now() modified_date
  FROM ( SELECT
         		product_name,
           	product_type,
           	unit_price
        	FROM PRODUCT_DELTA
  ) src
ON CONFLICT(product_name, product_type)
DO UPDATE
       SET
         unit_price   = excluded.unit_price,
         modified_date = excluded.modified_date;
