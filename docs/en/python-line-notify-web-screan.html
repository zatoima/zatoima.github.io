Catching Web Screen Changes and Sending LINE Notifications Using Python (pyautogui) and LINE Notify

Published on 2020/10/01 üìñ 1 min read
Python LINE Notify

A simple script that periodically checks for web screen changes and sends a LINE Notify notification when the expected screen appears.

Setting up LINE Notify itself is very easy.

[Super Simple] Trying out LINE Notify - Qiita https://qiita.com/iitenkida7/items/576a8226ba6584864d95

Originally I considered using Requests or Selenium, but no matter how I spoofed the user-agent for the target site, I kept getting access denied. So I used pyautogui to send a notification when the expected image appears on the web screen. pyautogui is even easier than UWSC!

The URLs in this script are dummy values.

import webbrowser
import datetime
import pyautogui
import time
import sys
import requests

def main():
    while True:
        time.sleep(300)
        check_web()
        if pos is not None:
            send_line_notify(sent_message)
            send_gazo()
        else:
            continue

def send_line_notify(notification_message):
    line_notify_token = 'xxxxxxxxxxxxxxxxxxxxxxxxxx'
    line_notify_api = 'https://notify-api.line.me/api/notify'
    headers = {'Authorization': f'Bearer {line_notify_token}'}
    data = {'message': f'message: {notification_message}'}
    requests.post(line_notify_api, headers = headers, data = data)

def send_gazo():
    line_notify_token = 'xxxxxxxxxxxxxxxxxxxxxxxx'
    line_notify_api = 'https://notify-api.line.me/api/notify'
    headers = {'Authorization': f'Bearer {line_notify_token}'}

    message = 'Screanshot for python script!'
    payload = {"message" :  message}
    files = {"imageFile":open('D:\\tool\\py\\screanshot.jpg','rb')}
    post = requests.post(line_notify_api ,headers = headers ,params=payload,files=files)

def check_web():

    # Open URL in browser
    webbrowser.open('https://www.google.com/')

    # Wait until fully displayed
    time.sleep(20)

    # Check if error screen is displayed using "pyautogui.locateOnScreen"
    pos_check
    pos_check=pyautogui.locateOnScreen('D:\\tool\\py\\access_denied.jpg', confidence=0.6)
    #print(pos_check)

    if pos_check is not None:
        print("Error - sleeping")
        # Record screenshot of error
        pyautogui.screenshot('D:\\tool\\py\\error.jpg')
        time.sleep(3600)

    # Fine-tune confidence
    # Check where on screen the image is located by coordinates
    pos=pyautogui.locateOnScreen('D:\\tool\\py\\google.jpg', confidence=0.6)
    #print(pos)

    # Set the message to send via LINE when the expected screen is displayed
    if pos is not None:
        global sent_message
        sent_message='LINE Notify Message xxxxxxxxxxxxxxxxxx'
        pyautogui.screenshot('D:\\tool\\py\\screanshot.jpg')

    else:
        print("No match")
        #sys.exit()

if __name__ == "__main__":
    main()
