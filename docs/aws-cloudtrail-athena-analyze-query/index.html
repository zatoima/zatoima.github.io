AthenaгҒ§CloudTrail гҒ®иЁји·ЎгӮ’еҲҶжһҗ

гҒ«е…¬й–Ӣ 2021/11/18 рҹ“– зҙ„1еҲҶ
AWS Athena CloudTrail

CloudTrailгҒ®гғһгғҚгғјгӮёгғЎгғігғҲгӮігғігӮҪгғјгғ«дёҠгҒ«гҒҜгҖҢAthenaгғҶгғјгғ–гғ«гӮ’дҪңжҲҗгҖҚгҒЁгҒ„гҒҶж©ҹиғҪгҒҢгҒӮгӮӢгҖӮгҒ“гҒ“гҒӢгӮүAthenaгҒ§жӨңзҙўгҒҢгҒ§гҒҚгӮӢгӮҲгҒҶгҒ«DDLгӮ’з”Ёж„ҸгҒ—гҒҰгҒҸгӮҢгҒҰгҒ„гӮӢгҒҢгҖҒгҒ“гӮҢгҒ«иҗҪгҒЁгҒ—з©ҙгҒҢгҒӮгҒЈгҒҹгҖӮS3дёҠгҒ®гғ•гӮЎгӮӨгғ«иҮӘдҪ“гҒҜж—Ҙд»ҳгҒ”гҒЁгҒ«ж јзҙҚгҒ•гӮҢгҒҰгғ‘гғјгғҶгӮЈгӮ·гғ§гғіеҢ–гҒ•гӮҢгҒҰгҒ„гӮӢгҒҢгҖҒгҒ“гҒ®DDLе®ҡзҫ©гҒҢгғ‘гғјгғҶгӮЈгӮ·гғ§гғігҒ«еҜҫеҝңгҒ—гҒҰгҒ„гҒӘгҒ„гҖӮдҪҷиЁҲгҒӘгӮігӮ№гғҲгӮ’зҷәз”ҹгҒ•гҒӣгҒӘгҒ„гҒҹгӮҒгҒ«гҖҒPartition ProjectionгҒ«еҜҫеҝңгҒ—гҒҹDDLгӮ’еҲҘйҖ”з”Ёж„ҸгҒҷгӮӢеҝ…иҰҒгҒҢгҒӮгӮӢгҖӮ

AthenaгҒ§гғҶгғјгғ–гғ«гӮ’дҪңжҲҗгҒҷгӮӢ

  • гҖҗе…ЁгғӘгғјгӮёгғ§гғіеҜҫеҝңгҖ‘CloudTrailгҒ®гғӯгӮ°гӮ’AthenaгҒ®Partition ProjectionгҒӘгғҶгғјгғ–гғ«гҒ§дҪңгӮӢ | DevelopersIO
  • AthenaгҒ®гғ‘гғјгғҶгӮЈгӮ·гғ§гғігғ—гғӯгӮёгӮ§гӮҜгӮ·гғ§гғігӮ’еҲ©з”ЁгҒ—гҒҰCloudTrailгҒ®гғӯгӮ°жӨңзҙўгӮ’гҒҷгӮӢ - гӮөгғјгғҗгғјгғҜгғјгӮҜгӮ№гӮЁгғігӮёгғӢгӮўгғ–гғӯгӮ° 
drop table cloudtrail_logs;

CREATE EXTERNAL TABLE cloudtrail_logs (
    eventVersion STRING,
    userIdentity STRUCT<
        type: STRING,
        principalId: STRING,
        arn: STRING,
        accountId: STRING,
        invokedBy: STRING,
        accessKeyId: STRING,
        userName: STRING,
        sessionContext: STRUCT<
            attributes: STRUCT<
                mfaAuthenticated: STRING,
                creationDate: STRING>,
            sessionIssuer: STRUCT<
                type: STRING,
                principalId: STRING,
                arn: STRING,
                accountId: STRING,
                userName: STRING>>>,
    eventTime STRING,
    eventSource STRING,
    eventName STRING,
    awsRegion STRING,
    sourceIpAddress STRING,
    userAgent STRING,
    errorCode STRING,
    errorMessage STRING,
    requestParameters STRING,
    responseElements STRING,
    additionalEventData STRING,
    requestId STRING,
    eventId STRING,
    resources ARRAY<STRUCT<
        arn: STRING,
        accountId: STRING,
        type: STRING>>,
    eventType STRING,
    apiVersion STRING,
    readOnly STRING,
    recipientAccountId STRING,
    serviceEventDetails STRING,
    sharedEventID STRING,
    vpcEndpointId STRING
)
COMMENT 'CloudTrail table for ${BucketName} bucket'
PARTITIONED BY (region string, date string)
ROW FORMAT SERDE 'com.amazon.emr.hive.serde.CloudTrailSerde'
STORED AS INPUTFORMAT 'com.amazon.emr.cloudtrail.CloudTrailInputFormat'
OUTPUTFORMAT 'org.apache.hadoop.hive.ql.io.HiveIgnoreKeyTextOutputFormat'
LOCATION 's3://z3-cloudtrail/AWSLogs/<Acclout ID>/CloudTrail/'
TBLPROPERTIES (
    'projection.enabled' = 'true',
    'projection.date.type' = 'date',
    'projection.date.range' = 'NOW-1YEARS,NOW',
    'projection.date.format' = 'yyyy/MM/dd',
    'projection.date.interval' = '1',
    'projection.date.interval.unit' = 'DAYS',
    'projection.region.type' = 'enum',
    'projection.region.values'='us-east-1,us-east-2,us-west-1,us-west-2,af-south-1,ap-east-1,ap-south-1,ap-northeast-2,ap-southeast-1,ap-southeast-2,ap-northeast-1,ca-central-1,eu-central-1,eu-west-1,eu-west-2,eu-south-1,eu-west-3,eu-north-1,me-south-1,sa-east-1',
    'storage.location.template' = 's3://z3-cloudtrail/AWSLogs/<Acclout ID>/CloudTrail/${region}/${date}',
    'classification'='cloudtrail',
    'compressionType'='gzip',
    'typeOfData'='file',
    'classification'='cloudtrail'
);

еҲҶжһҗгӮҜгӮЁгғӘгҒ®зҷәиЎҢ

жқұдә¬гғӘгғјгӮёгғ§гғігҒ§2021е№ҙ11жңҲгҒ«зҷәиЎҢгҒ•гӮҢгҒҹAPIгҒ§eventsourceгҖҒeventnameгҒ”гҒЁгҒ«йӣҶиЁҲгҒҷгӮӢ

SELECT
    awsregion,
    eventsource,
    eventname,
    COUNT(*) as cnt
FROM
    cloudtrail_logs
WHERE
    date>='2021/11/01' AND
    region = 'ap-northeast-1'    
GROUP BY
    awsregion,
    eventsource,
    eventname
ORDER BY cnt DESC
limit 10;

е®ҹиЎҢжҷӮй–“гӮӮе…ЁйҮҸжӨңзҙўгҒ«жҜ”гҒ№гҒҰй«ҳйҖҹеҢ–гҒ—гҒҰгҒ„гӮӢгҖӮгӮ№гӮӯгғЈгғігҒ—гҒҹгғҮгғјгӮҝгӮӮдёҖйғЁеҲҶгҒ®гҒҝгҖӮ

GitHubгҒ§з·ЁйӣҶгӮ’жҸҗжЎҲ
вҶҗ еүҚгҒ®иЁҳдәӢ DMBOKеӢүеј·гҒ®ж®ҙгӮҠжӣёгҒҚгғЎгғўпјҲеҸӮз…§гғҮгғјгӮҝгҒЁгғһгӮ№гӮҝгғҮгғјгӮҝпјү ж¬ЎгҒ®иЁҳдәӢ вҶ’ VSCodeгҒ§гғҮгғ•гӮ©гғ«гғҲгҒ®гӮ·гғігӮҝгғғгӮҜгӮ№гғҸгӮӨгғ©гӮӨгғҲгӮ’иЁӯе®ҡгҒҷгӮӢ
е…ұжңү: B!

й–ўйҖЈгҒ—гҒҰгҒ„гӮӢгҒӢгӮӮгҒ—гӮҢгҒӘгҒ„иЁҳдәӢ

  • AthenaгӮ’CLIгғҷгғјгӮ№гҒ§е®ҹиЎҢгҒҷгӮӢпјҲAWS CLIгҖҒathenacliпјү
  • ALB(Application Load Balancer)гҒ®гғӯгӮ°гӮ’AthenaгҒ§еҲҶжһҗ
  • Redshift SpectrumгҒӢгӮүGlue Data CatalogгҒ®гғҶгғјгғ–гғ«гӮ’еҸӮз…§гҒҷгӮӢ
  • awswranglerгӮ’дҪҝз”ЁгҒ—гҒҰCSV to Parquet
  • Redshift SpectrumгҒ®гғҷгӮ№гғҲгғ—гғ©гӮҜгғҶгӮЈгӮ№
